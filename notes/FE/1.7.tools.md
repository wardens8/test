```javascript
/**
 * 格式化日期时间函数
 * formatDateTime.date() - 格式化日期，直接调用，如：2024年03月19日 星期二
 * formatDateTime.time() - 格式化时间，直接调用，如：14:30:45
 * formatDateTime.all() - 格式化日期和时间，直接调用，如：2024年03月19日 星期二 14:30:45
 */
const formatDateTime = {
  // 格式化日期
  date: function () {
    const now = new Date()
    const year = now.getFullYear().toString()
    const month = (now.getMonth() + 1).toString().padStart(2, '0')
    const date = now.getDate().toString().padStart(2, '0')
    const week = ['日', '一', '二', '三', '四', '五', '六']
    const day = now.getDay()
    return `${year}年${month}月${date}日 星期${week[day]}`
  },
  // 格式化时间
  time: function () {
    const now = new Date()
    const hour = now.getHours().toString().padStart(2, '0')
    const minute = now.getMinutes().toString().padStart(2, '0')
    const second = now.getSeconds().toString().padStart(2, '0')
    return `${hour}:${minute}:${second}`
  },
  // 格式化日期和时间
  all: function () {
    const now = new Date()
    const year = now.getFullYear().toString()
    const month = (now.getMonth() + 1).toString().padStart(2, '0')
    const date = now.getDate().toString().padStart(2, '0')
    const week = ['日', '一', '二', '三', '四', '五', '六']
    const day = now.getDay()
    const hour = now.getHours().toString().padStart(2, '0')
    const minute = now.getMinutes().toString().padStart(2, '0')
    const second = now.getSeconds().toString().padStart(2, '0')
    return `${year}年${month}月${date}日 星期${week[day]} ${hour}:${minute}:${second}`
  },
}
```

```javascript
/**
 * 格式化日期时间函数
 * @param {Date|string|number} date - 日期对象、日期字符串或时间戳（可选，默认为当前时间）
 * @param {string} format - 格式化模板（可选，默认为 'YYYY年MM月DD日 星期WWW HH:mm:ss'）
 * @returns {string} 格式化后的日期字符串
 */
function formatDateTime(date = new Date(), format = 'YYYY年MM月DD日 星期WWW HH:mm:ss') {
  // 将输入转换为 Date 对象
  const now = new Date(date)
  // 验证日期是否有效
  if (isNaN(now.getTime())) {
    console.error('无效的日期格式')
    return 'Invalid Date'
  }
  const tokens = {
    // 年
    YYYY: now.getFullYear(),
    YY: now.getFullYear().toString().slice(-2),
    // 月
    MM: (now.getMonth() + 1).toString().padStart(2, '0'),
    M: now.getMonth() + 1,
    // 日
    DD: now.getDate().toString().padStart(2, '0'),
    D: now.getDate(),
    // 时（24小时制）
    HH: now.getHours().toString().padStart(2, '0'),
    H: now.getHours(),
    // 时（12小时制）
    hh: (now.getHours() % 12 || 12).toString().padStart(2, '0'),
    h: now.getHours() % 12 || 12,
    // 分
    mm: now.getMinutes().toString().padStart(2, '0'),
    m: now.getMinutes(),
    // 秒
    ss: now.getSeconds().toString().padStart(2, '0'),
    s: now.getSeconds(),
    // 毫秒
    SSS: now.getMilliseconds().toString().padStart(3, '0'),
    SS: now.getMilliseconds().toString().padStart(3, '0').slice(0, 2),
    S: now.getMilliseconds().toString().padStart(3, '0').slice(0, 1),
    // 星期
    WWW: ['日', '一', '二', '三', '四', '五', '六'][now.getDay()],
    WW: ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'][now.getDay()],
    W: ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'][now.getDay()],
    // 上午/下午
    A: now.getHours() < 12 ? 'AM' : 'PM',
    a: now.getHours() < 12 ? 'am' : 'pm',
    '上午/下午': now.getHours() < 12 ? '上午' : '下午',
  }
  // 替换格式字符串中的标记
  return format.replace(/YYYY|YY|MM|M|DD|D|HH|H|hh|h|mm|m|ss|s|SSS|SS|S|WWW|WW|W|A|a|上午\/下午/g, (match) => tokens[match] || match)
}

// ==================== 常用格式化模板 ====================
const formats = {
  // 默认格式
  default: 'YYYY年MM月DD日 星期WWW HH:mm:ss', // 2024年03月19日 星期二 14:30:45
  // 标准格式
  standard: 'YYYY-MM-DD HH:mm:ss', // 2024-03-19 14:30:45
  date: 'YYYY-MM-DD', // 2024-03-19
  time: 'HH:mm:ss', // 14:30:45
  // 中文格式
  chinese: 'YYYY年MM月DD日 HH时mm分ss秒', // 2024年03月19日 14时30分45秒
  chineseShort: 'YYYY年M月D日', // 2024年3月19日
  // 带星期
  withWeekday: 'YYYY-MM-DD 星期WW', // 2024-03-19 星期二
  withWeekEn: 'YYYY-MM-DD (W)', // 2024-03-19 (Tuesday)
  // 12小时制
  hour12: 'YYYY-MM-DD hh:mm:ss A', // 2024-03-19 02:30:45 PM
  // 文件命名常用
  filename: 'YYYYMMDD_HHmmss', // 20240319_143045
  filenameMs: 'YYYYMMDD_HHmmss_SSS', // 20240319_143045_123
  // 日志格式
  log: 'YYYY-MM-DD HH:mm:ss.SSS', // 2024-03-19 14:30:45.123
  // 简洁格式

  slash: 'YYYY/M/D', // 2024/3/19
```

```javascript
// 定义
function updateClockHands(hourHand, minuteHand, secondHand) {
  const now = new Date()
  const hours = now.getHours() % 12 // 转为12小时制
  const minutes = now.getMinutes()
  const seconds = now.getSeconds()
  // 计算旋转角度（单位：deg）
  // 时针：12小时=360° → 每小时30°，且随分钟偏移（30°/60=0.5° per分钟）
  const hourDeg = hours * 30 + minutes * 0.5
  // 分针：60分钟=360° → 每分钟6°，随秒偏移（6°/60=0.1° per秒）
  const minuteDeg = minutes * 6 + seconds * 0.1
  // 秒针：60秒=360° → 每秒6°
  const secondDeg = seconds * 6
  // 仅修改CSS样式实现转动，无任何DOM增删/属性修改
  hourHand.style.transform = `rotate(${hourDeg}deg)`
  minuteHand.style.transform = `rotate(${minuteDeg}deg)`
  secondHand.style.transform = `rotate(${secondDeg}deg)`
}
// 使用
updateClockHands(shi, fen, miao)
setInterval(() => {
  updateClockHands(shi, fen, miao)
}, 1000)
```

```javascript
function initTodos() {
  const todoInput = document.getElementById('todo-input')
  const addButton = document.getElementById('add-button')
  const removeButton = document.getElementById('remove-button')
  const todoList = document.getElementById('todo-list')

  // 从本地存储中获取待办事项列表
  let todos = JSON.parse(localStorage.getItem('Todos')) || []

  // 渲染待办事项列表
  function renderTodos() {
    todoList.innerHTML = ''
    todos.forEach((todo, index) => {
      const li = document.createElement('li')
      // 设置不同样式 完成样式:未完成样式
      li.style.textDecoration = todo.done ? 'line-through' : 'none'
      li.style.color = todo.done ? '#ccc' : '#000'
      li.textContent = todo.text
      li.dataset.index = index

      // 添加删除按钮
      const deleteButton = document.createElement('a')
      deleteButton.href = 'javascript:void(0);'
      deleteButton.textContent = '删除'
      deleteButton.style.marginLeft = '10px'
      deleteButton.style.color = todo.done ? '#ccc' : '#00e'
      deleteButton.addEventListener('click', () => {
        todos.splice(index, 1)
        localStorage.setItem('Todos', JSON.stringify(todos))
        renderTodos()
      })
      li.append(deleteButton)

      // 添加标记完成按钮
      const doneButton = document.createElement('input')
      doneButton.type = 'checkbox'
      doneButton.checked = todo.done
      doneButton.addEventListener('change', () => {
        todo.done = !todo.done
        localStorage.setItem('Todos', JSON.stringify(todos))
        renderTodos()
      })
      li.prepend(doneButton)

      // 把事项添加到列表中
      todoList.prepend(li)
    })
  }

  // 添加待办事项
  addButton.addEventListener('click', () => {
    const text = todoInput.value.trim()
    if (text) {
      todos.push({ text, done: false })
      localStorage.setItem('Todos', JSON.stringify(todos))
      renderTodos()
      todoInput.value = ''
    }
  })

  // 清空待办事项
  removeButton.addEventListener('click', () => {
    todos = []
    localStorage.setItem('Todos', JSON.stringify(todos))
    renderTodos()
  })

  // 初始化渲染待办事项列表
  renderTodos()
}

// 初始化待办事项
initTodos()
```

```javascript
function initWeather() {
  // get获取天气
  function getWeather() {
    const tbody = document.querySelector('tbody')
    const xhr = new XMLHttpRequest()
    const apiUrl = 'https://api.seniverse.com/v3/weather/daily.json?key=SCYrvkytJze9qyzOh&location=rizhao&language=zh-Hans&unit=c'
    // 配置请求（方法、地址、是否异步）

    // 设置响应类型（可选，如 'json'、'text'）

                <td>${daily[i].precip}%</td>

                <td>${daily[i].rainfall}级</td>
                <td>${daily[i].wind_speed}km/h</td>
              </tr>
          `,
          )
        }
      } else {
        // 请求失败
        console.error('请求失败，状态码：', xhr.status)
```

我们需要借助一个 `api` 叫做 `SpeechSynthesisUtterance` 来实现

- `SpeechSynthesisUtterance` 是一个接口，用于语音合成，它允许我们创建一个语音合成实例，并设置其属性，如文本、音量、语速等

```javascript

speakAudio.text = 'Hello, world!'
// 获取并设置语音合成的音量
speakAudio.volume = 1
// 获取并设置语音合成的语速
speakAudio.tate = 1
// 获取并设置语音合成的音调
speakAudio.pitch = 1
// 获取并设置语音合成的语言
speakAudio.lang = 'en-US'
```

```javascript
function getRGBA(item) {
  const col = getComputedStyle(item).backgroundColor
  // console.log(col) // 打印颜色完整值
  const q = col.match(/[\d+.]+/g)
  const r = col.match(/[\d+.]+/g)[0]
  const g = col.match(/[\d+.]+/g)[1]
  const b = col.match(/[\d+.]+/g)[2]
  const a = col.match(/[\d+.]+/g)[3]
  // console.log(r, g, b, a) // 分别打印rgba值
  const L = (0.299 * r + 0.587 * g + 0.114 * b) / 255
  if (L > 0.65) {
    div.innerHTML = '浅色'
    div.style.color = 'black'
  } else {
    div.innerHTML = '深色'
    div.style.color = 'white'
  }
}
```
